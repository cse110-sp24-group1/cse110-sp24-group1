One and Done

Welcome to One and Done, a feature-rich journal app built with HTML, CSS, and JavaScript. This app allows you to create notes and folders, use markdown for formatting, and manage tasks efficiently.

Learn More About our Team!

Features

  • Organized Note and Folder Creation: Easily create and manage your notes and folders to keep your thoughts and tasks well-organized.
  • Markdown Support: Utilize powerful markdown capabilities to format your notes with ease, adding structure and emphasis.
  • Efficient Task List Management: Keep track of your tasks and to-dos effortlessly with a dedicated task list management system.
  • Dual Theme Options: Switch seamlessly between dark mode and light mode to suit your preferences and enhance your writing experience.
  • Persistent Data Management: Rely on local storage to securely manage your data, ensuring your notes and tasks are always saved.
  • Powerful Search Functionality: Quickly find specific notes and tasks with a robust search bar, enhancing productivity and accessibility.

Jest and Puppeteer

  • For testing purposes, we use Jest and Puppeteer.
  • Here's how to set up and run the tests:

Setup

  • Install Dependencies: 
    npm install --save-dev jest babel-jest @babel/core @babel/preset-env puppeteer jest-puppeteer
  • Scripts: Add the following scripts to your package.json: 
    {
        "devDependencies": {
          "@babel/core": "^7.24.5",
          "@babel/preset-env": "^7.24.5",
          "babel-jest": "^29.7.0",
          "jest": "^29.7.0",
          "jest-puppeteer": "^10.0.0",
          "puppeteer": "^22.7.1"
        },
        "scripts": {
          "test": "jest --maxWorkers=1"
        },
        "jest": {
          "preset": "jest-puppeteer",
          "verbose": true
        }
      }

Writing Tests

Here are some example test cases for notes, folders, and tasks functionalities using Jest and Puppeteer, note that we first run live server and place in await page.goto() :

Testing Notes
const puppeteer = require('puppeteer');
  
  describe('Notes Functionality Tests', () => {
    let browser;
    let page;
  
    beforeAll(async () => {
      browser = await puppeteer.launch({
        headless: false,
        slowMo: 50,
      });
      page = await browser.newPage();
      await page.goto('http://127.0.0.1:5500/src/index.html');
    });
  
    afterAll(async () => {
      await browser.close();
    });
  
    test('should recognize that there are no notes', async () => {
      await page.waitForSelector('main');
      const notesCount = await page.$$eval('.note', notes => notes.length);
      expect(notesCount).toBe(0);
    });
  
    test('should create a new note with a title and label', async () => {
      await page.waitForSelector('#new-note-button');
      await page.click('#new-note-button');
      await page.waitForSelector('#note-title');
      await page.type('#note-title', 'Hello World Skeleton');
      await page.select('#note-label', 'code-snippets');
      await page.click('.create-button');
      await page.waitForSelector('.note');
      const noteTitle = await page.$eval('.note-title h3', el => el.innerText);
      const noteLabel = await page.$eval('.note-content', el => el.id);
      expect(noteTitle).toBe('Hello World Skeleton');
      expect(noteLabel).toBe('code-snippets');
    }, 20000);
  
    test('should edit an existing note title', async () => {
      jest.setTimeout(30000);
      await page.waitForSelector('.note');
      await page.click('.note');
      await page.waitForSelector('.note-modal');
      const editTitle = await page.$('.edit-modal-title h2');
      await editTitle.click({ clickCount: 3 });
      await page.keyboard.press('Backspace');
      await editTitle.type('Hello World Skeleton C++');
      await page.evaluate(() => {
        document.querySelector('.save-button').click();
      });
      await page.waitForFunction(
        () => document.querySelector('.note-title h3').innerText.includes('Hello World Skeleton C++')
      );
      const noteTitle = await page.$eval('.note-title h3', el => el.innerText);
      expect(noteTitle).toBe('Hello World Skeleton C++');
    }, 30000);
  
    test('should delete an existing note', async () => {
      jest.setTimeout(30000);
      await page.waitForSelector('.note');
      const noteElement = await page.$('.note');
      const boundingBox = await noteElement.boundingBox();
      const x = boundingBox.x + boundingBox.width / 2;
      const y = boundingBox.y + boundingBox.height / 2;
      await page.mouse.move(x, y);
      await delay(1000);
      await page.evaluate(() => {
        const deleteButton = document.querySelector('.note .delete');
        if (deleteButton) {
          deleteButton.style.display = 'block';
        }
      });
      const deleteButton = await page.$('.note .delete');
      await page.mouse.move(deleteButton.boundingBox().x, deleteButton.boundingBox().y);
      await delay(500);
      await deleteButton.click();
      await page.waitForSelector('.yes-button');
      await page.click('.yes-button');
      const notesCount = await page.$$eval('.note', notes => notes.length);
      expect(notesCount).toBe(0);
    }, 30000);
  });

Testing Tasks
const puppeteer = require('puppeteer');
  
  describe('Task List Functionality Tests', () => {
    let browser;
    let page;
  
    beforeAll(async () => {
      browser = await puppeteer.launch({
        headless: false,
        slowMo: 50,
      });
      page = await browser.newPage();
      await page.goto('http://127.0.0.1:5500/src/tasks.html');
    });
  
    afterAll(async () => {
      await browser.close();
    });
  
    test('Checking that no tasks are loaded initially', async () => {
      const tasks = await page.evaluate(() => {
        const taskListElement = document.querySelector('task-list');
        const shadowRoot = taskListElement.shadowRoot;
        const taskContainer = shadowRoot.querySelector('.task-container');
        return Array.from(taskContainer.querySelectorAll('.task-item'));
      });
      expect(tasks.length).toBe(0);
    });
  
    test('Checking that user can add a new task', async () => {
      await page.type('#new-task-input', 'Groceries');
      await page.click('#task-form button[type="submit"]');
      await page.waitForSelector('#modal', { visible: true });
      await page.type('#task-description', 'Fruits and veggies');
      await page.evaluate(() => {
        document.querySelector('#task-due-date').value = '2024-05-20';
      });
      await page.select('#task-label', 'Personal');
      await page.click('.sub-button');
      await page.waitForSelector('#modal', { hidden: true });
      const taskExists = await page.evaluate(() => {
        const taskListElement = document.querySelector('task-list');
        const shadowRoot = taskListElement.shadowRoot;
        const taskContainer = shadowRoot.querySelector('.task-container');
        return taskContainer.querySelectorAll('.task-item').length === 1;
      });
      expect(taskExists).toBeTruthy();
    }, 20000);
  });

Running the Tests

To run the tests, simply use the following command in your terminal:

npm test

This command will execute all test files with the .test.js extension and report the results in the terminal. If you would like to test a file individually you can just do

 npm test -- (filename).test.js
